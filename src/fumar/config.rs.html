8/// Persistent configuration loaded from `~/.config/fumar/config.kdl`.
9#[derive(Debug, Clone)]
10pub struct Config {
11    /// Default UI mode: "tui", "cli", or "gui".
12    pub mode: String,
13    /// Enable Discord Rich Presence.
14    pub discord: bool,
15    /// BLE scan timeout in seconds.
16    pub scan_timeout: u64,
17    /// Default target temperature in Celsius.
18    pub default_temp: f32,
19    /// Auto-connect to first found device.
20    pub auto_connect: bool,
21}
22
23impl Default for Config {
24    fn default() -> Self {
25        Self {
26            mode: "tui".into(),
27            discord: false,
28            scan_timeout: 10,
29            default_temp: 180.0,
30            auto_connect: false,
31        }
32    }
33}
34
35impl Config {
36    /// Path to the config file.
37    pub fn config_path() -> Result<PathBuf> {
38        let base = dirs_fallback();
39        Ok(base.join("fumar").join("config.kdl"))
40    }
41
42    /// Load config from disk, creating with interactive prompts if missing.
43    pub fn load() -> Result<Self> {
44        let path = Self::config_path()?;
45
46        if !path.exists() {
47            debug!("Config not found at {path:?}, starting first-run setup");
48            let config = Self::interactive_setup()?;
49            config.save()?;
50            eprintln!("\nConfig saved to {}\n", path.display());
51            return Ok(config);
52        }
53
54        let content = std::fs::read_to_string(&path)
55            .with_context(|| format!("Failed to read config from {path:?}"))?;
56
57        Self::parse(&content)
58    }
59
60    /// Interactive first-run setup — prompts user for each setting.
61    fn interactive_setup() -> Result<Self> {
62        use std::io::{self, Write};
63
64        eprintln!("Welcome to fumar! Let's set up your config.\n");
65
66        let stdin = io::stdin();
67        let mut lines = stdin.lock();
68
69        let mut config = Self::default();
70
71        // Mode
72        eprintln!("Default UI mode:");
73        eprintln!("  [1] TUI  (terminal interface, default)");
74        eprintln!("  [2] CLI  (command-line only)");
75        eprintln!("  [3] GUI  (graphical, requires GTK4)");
76        eprint!("Choice [1]: ");
77        io::stderr().flush()?;
78        config.mode = match read_line(&mut lines).as_str() {
79            "2" => "cli".into(),
80            "3" => "gui".into(),
81            _ => "tui".into(),
82        };
83        eprintln!();
84
85        // Discord
86        eprint!("Enable Discord Rich Presence? [y/N]: ");
87        io::stderr().flush()?;
88        config.discord = read_line(&mut lines).to_lowercase().starts_with('y');
89        eprintln!();
90
91        // Scan timeout
92        eprint!("BLE scan timeout in seconds [10]: ");
93        io::stderr().flush()?;
94        let input = read_line(&mut lines);
95        if let Ok(val) = input.parse::<u64>() {
96            config.scan_timeout = val.clamp(1, 120);
97        }
98        eprintln!();
99
100        // Default temp
101        eprint!("Default target temperature in °C [180]: ");
102        io::stderr().flush()?;
103        let input = read_line(&mut lines);
104        if let Ok(val) = input.parse::<f32>() {
105            config.default_temp = val.clamp(40.0, 230.0);
106        }
107        eprintln!();
108
109        // Auto-connect
110        eprint!("Auto-connect to first device found? [y/N]: ");
111        io::stderr().flush()?;
112        config.auto_connect = read_line(&mut lines).to_lowercase().starts_with('y');
113        eprintln!();
114
115        eprintln!("Configuration complete!\n");
116
117        Ok(config)
118    }

120    /// Parse a KDL string into a Config.
121    fn parse(content: &str) -> Result<Self> {
122        let doc: KdlDocument = content.parse().context("Failed to parse config.kdl")?;
123
124        let mut config = Self::default();
125
126        // Parse "mode" node
127        if let Some(node) = doc.get("mode")
128            && let Some(val) = node.get(0).and_then(|v| v.as_string())
129        {
130            config.mode = val.to_string();
131        }
132
133        // Parse "discord" node
134        if let Some(node) = doc.get("discord")
135            && let Some(val) = node.get(0).and_then(|v| v.as_bool())
136        {
137            config.discord = val;
138        }
139
140        // Parse "scan_timeout" node
141        if let Some(node) = doc.get("scan_timeout")
142            && let Some(val) = node.get(0).and_then(|v| v.as_integer())
143        {
144            config.scan_timeout = val.clamp(1, 120) as u64;
145        }
146
147        // Parse "default_temp" node
148        if let Some(node) = doc.get("default_temp")
149            && let Some(val) = node.get(0).and_then(|v| v.as_float())
150        {
151            config.default_temp = (val as f32).clamp(40.0, 230.0);
152        }
153
154        // Parse "auto_connect" node
155        if let Some(node) = doc.get("auto_connect")
156            && let Some(val) = node.get(0).and_then(|v| v.as_bool())
157        {
158            config.auto_connect = val;
159        }
160
161        debug!("Loaded config: {config:?}");
162        Ok(config)
163    }

165    /// Save config to disk as KDL.
166    pub fn save(&self) -> Result<()> {
167        let path = Self::config_path()?;
168
169        if let Some(parent) = path.parent() {
170            std::fs::create_dir_all(parent)
171                .with_context(|| format!("Failed to create config dir {parent:?}"))?;
172        }
173
174        let content = self.to_kdl();
175        std::fs::write(&path, content)
176            .with_context(|| format!("Failed to write config to {path:?}"))?;
177
178        debug!("Saved config to {path:?}");
179        Ok(())
180    }
181
182    /// Serialize config to KDL string.
183    fn to_kdl(&self) -> String {
184        let mut doc = KdlDocument::new();
185
186        let mut mode_node = KdlNode::new("mode");
187        mode_node.push(self.mode.clone());
188        doc.nodes_mut().push(mode_node);
189
190        let mut discord_node = KdlNode::new("discord");
191        discord_node.push(KdlValue::Bool(self.discord));
192        doc.nodes_mut().push(discord_node);
193
194        let mut timeout_node = KdlNode::new("scan_timeout");
195        timeout_node.push(self.scan_timeout as i128);
196        doc.nodes_mut().push(timeout_node);
197
198        let mut temp_node = KdlNode::new("default_temp");
199        temp_node.push(self.default_temp as f64);
200        doc.nodes_mut().push(temp_node);
201
202        let mut auto_node = KdlNode::new("auto_connect");
203        auto_node.push(KdlValue::Bool(self.auto_connect));
204        doc.nodes_mut().push(auto_node);
205
206        // Auto-format for readability
207        doc.format();
208
209        doc.to_string()
210    }
211}

213/// Read a trimmed line from stdin, returning empty string on EOF.
214fn read_line(reader: &mut impl BufRead) -> String {
215    let mut buf = String::new();
216    if reader.read_line(&mut buf).is_ok() {
217        buf.trim().to_string()
218    } else {
219        String::new()
220    }
221}

223/// Get config directory, falling back if `dirs` crate not available.
224fn dirs_fallback() -> PathBuf {
225    if let Some(dir) = std::env::var_os("XDG_CONFIG_HOME")
226        .filter(|s| !s.is_empty())
227        .map(PathBuf::from)
228    {
229        return dir;
230    }
231
232    if let Some(home) = std::env::var_os("HOME") {
233        return PathBuf::from(home).join(".config");
234    }
235
236    // Windows fallback
237    if let Some(appdata) = std::env::var_os("APPDATA") {
238        return PathBuf::from(appdata);
239    }
240
241    PathBuf::from(".")
242}
